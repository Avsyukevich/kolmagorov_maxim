\subsection{Перевод в двоичную систему счисления и обратно методом №1}
\subsubsection{Перевод числа 68 в двоичную систему счисления}
\begin{enumerate}
  \item $\frac{69}{2}=34\frac{1}{2}$
  \item $\frac{34}{2}=17(\frac{0}{2})$
  \item $\frac{17}{2}=8\frac{1}{2}$
  \item $\frac{8}{2}=4(\frac{0}{2})$
  \item $\frac{4}{2}=2(\frac{0}{2})$
  \item $\frac{2}{2}=1(\frac{0}{2})$
Из данных расчётов следует, что $68_10=1000101_2$.
\end{enumerate}
